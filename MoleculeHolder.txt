public class MoleculeHolder {
    
    private static HashMap<Character, Integer> molecules;
    private static int totalMolecules = 0;
    
    static
    {
        molecules = new HashMap<Character, Integer>();
        molecules.put("A", 0);
        molecules.put("B", 0);
        molecules.put("C", 0);
        molecules.put("D", 0);
        molecules.put("E", 0);
    }

    public MoleculeHolder() {
        
    }
    
    public int getMolecule(char molecule) {
        
        return molecules.get(molecule);
    }
    
    public int addMolecule(char molecule) {
        if (molecules.get(molecule).equals(null)) {
            return -1
        } else if (totalMolecules >= 10) {
            System.err.println("You are already holding 10 molecules.")
            return -1;
        }
            totalMolecules++;
            return molecules.put(molecule, molecules.get(molecule)++);
    }
    
    public int useMolecule(char molecule) {
        if (molecules.get(molecule) - 1 >= 0) {
            totalMolecules--;
            return molecules.put(molecule, molecules.get(molecule)--);
        } else {
            System.err.println("There are no more " + molecule + " molecules to use.")
            return -1;
        }
    }
    
}
