public class SampleHolder {
    
    private static HashMap<Character, Integer> samples;
    private static int totalSamples = 0;
    
    static
    {
        samples = new HashMap<Integer, Sample;
  

    public SampleHolder(int sampleCount) {
        
    }
    
    public int getSample(char sample) {
        
        return samples.get(sample);
    }
    
    public int addSample(char sample) {
        if (samples.get(sample).equals(null)) {
            return -1
        } else if (totalSamples >= 10) {
            System.err.println("You are already holding 10 samples.")
            return -1;
        }
            totalSamples++;
            return samples.put(sample, samples.get(sample)++);
    }
    
    public int useSample(char sample) {
        if (samples.get(sample) - 1 >= 0) {
            totalSamples--;
            return samples.put(sample, samples.get(sample)--);
        } else {
            System.err.println("There are no more " + sample + " samples to use.")
            return -1;
        }
    }
    
}
